@code {
    [Parameter] public int Id { get; set; }
    [Parameter] public bool MostrarEncabezado { get; set; } = true;

    ClienteDetailDto? cliente;
    bool isLoading = true;

    bool modalTarifas;
    bool modalEstadoCuenta;
    bool modalMedidores;

    protected override async Task OnParametersSetAsync()
    {
        isLoading = true;
        cliente = await ClientesClient.ObtenerPorIdAsync(Id);
        isLoading = false;
    }

    void BackToList() => Navigation.NavigateTo("/clientes");

    string GetInitials(string? name)
        => string.IsNullOrWhiteSpace(name) ? "C" : name.Substring(0, 1).ToUpper();

    private static string MostrarTexto(string? valor)
        => string.IsNullOrWhiteSpace(valor) ? "—" : valor;

    private static string MostrarFecha(DateTime? fecha)
        => fecha.HasValue ? fecha.Value.ToString("d") : "—";

    private static string MostrarBooleano(bool? valor)
        => valor.HasValue ? (valor.Value ? "Sí" : "No") : "—";

    private static string MostrarDecimal(double? valor)
        => valor.HasValue ? valor.Value.ToString("N2") : "—";

    private static string MostrarAbogado(ClienteDetailDto cliente)
        => !string.IsNullOrWhiteSpace(cliente.AbogadoNombre)
            ? cliente.AbogadoNombre!
            : (cliente.AbogadoId?.ToString() ?? "—");

    // ---------- Visual helpers (badges / chips) ----------
    private static string MostrarObj(object? valor)
        => valor is null ? "—" : (valor.ToString() ?? "—");

    private static string BadgeClassForText(string? valor)
        => string.IsNullOrWhiteSpace(valor) || valor.Trim() == "—"
            ? "kv-badge kv-badge--muted"
            : "kv-badge kv-badge--info";

    private static string BadgeClassForObj(object? valor)
        => BadgeClassForText(MostrarObj(valor));

    private static string BadgeClassForDate(DateTime? valor)
        => valor.HasValue ? "kv-badge kv-badge--info" : "kv-badge kv-badge--muted";

    private static string BadgeClassForDecimal(double? valor)
        => valor.HasValue ? "kv-badge kv-badge--info" : "kv-badge kv-badge--muted";

    private static string BadgeClassForBool(bool? valor)
        => valor == true ? "kv-badge kv-badge--ok"
         : valor == false ? "kv-badge kv-badge--no"
         : "kv-badge kv-badge--muted";

    private static string BoolIcon(bool? valor)
        => valor == true ? "bi bi-check-circle-fill"
         : valor == false ? "bi bi-x-circle-fill"
         : "bi bi-dash-circle";

}
